import { useEffect } from "react"

export const Redirect = () => {
  useEffect(() => {
    {/* Only perform the redirect on the published site (no need in the preview) */}
    if (window.location.pathname?.endsWith('/docs/api')) {
      window.location.href = "https://cal.com/docs/enterprise-features/api";
    }
  }, [])
  
  return <></>
}

<Redirect />
